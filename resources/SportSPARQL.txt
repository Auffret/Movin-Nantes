PREFIX : <http://dbpedia.org/resource/>
PREFIX lod: <http://lodpaddle.univ-nantes.fr/>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>
PREFIX dbp: <http://dbpedia.org/property/>
PREFIX geo: <http://www.w3.org/2003/01/geo/wgs84_pos#>
select distinct ?name ?adresse ?latitude ?longitude
where {
    {
    ?subject rdf:type lod:sport_loisir;
        foaf:name ?name;
        dbp:location ?adresse;
        geo:lat ?latitude;
        geo:long ?longitude.
    {?subject lod:statut "Ville de Nantes"^^<http://www.w3.org/2001/XMLSchema#string>} UNION {?subject lod:statut "Etat"^^<http://www.w3.org/2001/XMLSchema#string>}.
    }
    UNION{
    ?subject rdf:type lod:vie_pratique;
        foaf:name ?name;
        lod:typeLabel "Parc"^^<http://www.w3.org/2001/XMLSchema#string>;
        dbp:town :Nantes;
        dbp:location ?adresse;
        geo:lat ?latitude;
        geo:long ?longitude.
    }
}
